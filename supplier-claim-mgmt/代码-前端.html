前端代码逻辑 · F-CLAIM-05

本轮最小场景开发的服务商索赔单「查看」功能（列表 + 详情）前端实现。技术栈 React + TypeScript + Vite + Ant Design。按「路由 → service（Mock/真实切换）→ 页面 → 类型/工具」的调用链组织。

// 服务商索赔单 service：VITE_USE_MOCK=true 时走本地 Mock，否则走真实后端// 接口路径权威源：docs/api-contracts/接口文档-服务商索赔单.md S2.8（GET /claim/order/{id}）/ S2.9（GET /claim/order/page）import type { PageResult, Result } from '../types/common'import type { ClaimOrderDetail, ClaimOrderListItem, ClaimOrderPageQuery } from '../types/claimOrder'import { getMockClaimOrderDetail, getMockClaimOrderPage } from '../mocks'import api from './api' const useMock = import.meta.env.VITE_USE_MOCK === 'true' export async function fetchClaimOrderPage(  query: ClaimOrderPageQuery): Promise<Result<PageResult<ClaimOrderListItem>>> {  if (useMock) {    return getMockClaimOrderPage(query)  }  const { data } = await api.get<Result<PageResult<ClaimOrderListItem>>>('/claim/order/page', {    params: query,  })  return data} export async function fetchClaimOrderDetail(id: string): Promise<Result<ClaimOrderDetail>> {  if (useMock) {    return getMockClaimOrderDetail(id)  }  const { data } = await api.get<Result<ClaimOrderDetail>>(`/claim/order/${id}`)  return data}
